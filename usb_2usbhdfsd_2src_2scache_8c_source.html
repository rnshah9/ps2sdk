49static void initRecords(cache_set* cache)
50{
51 unsigned int i;
52
53 for (i = 0; i < CACHE_SIZE; i++)

55 cache->rec[i].sector = 0xFFFFFFF0;

63//---------------------------------------------------------------------------
64/* search cache records for the sector number stored in cache
65 returns cache record (slot) number
66 */
67static int getSlot(cache_set* cache, unsigned int sector) {
68 int i;
69
70 for (i = 0; i < CACHE_SIZE; i++) {
71 if (sector >= cache->rec[i].sector && sector < (cache->rec[i].sector + cache->indexLimit)) {
72 return i;
73 }
74 }
75 return -1;
76}

78//---------------------------------------------------------------------------
79/* search cache records for the sector number stored in cache */
80static int getIndexRead(cache_set* cache, unsigned int sector) {
81 unsigned int i;
82 int index =-1;
83
84 for (i = 0; i < CACHE_SIZE; i++) {
85 if (sector >= cache->rec[i].sector && sector < (cache->rec[i].sector + cache->indexLimit)) {
86 if (cache->rec[i].tax < 0) cache->rec[i].tax = 0;
87 cache->rec[i].tax +=2;
88 index = i;
89 }
90 cache->rec[i].tax--; //apply tax penalty
91 }
92 if (index < 0)
93 return index;
94 else
95 return ((index * cache->indexLimit) + (sector - cache->rec[index].sector));
96}

98//---------------------------------------------------------------------------
99/* select the best record where to store new sector */
100static int getIndexWrite(cache_set* cache, unsigned int sector) {
101 int ret;
102 int minTax = 0x0FFFFFFF;
103 unsigned int i, index = 0;
104
105 for (i = 0; i < CACHE_SIZE; i++) {
106 if (cache->rec[i].tax < minTax) {
107 index = i;
108 minTax = cache->rec[i].tax;
109 }
110 }
111
112 //this sector is dirty - we need to flush it first
113 ret = scache_flushSector(cache, index);
114 if (ret != 1)
115 return ret;
116
117 cache->rec[index].tax +=2;
118 cache->rec[index].sector = sector;
119
120 return index * cache->indexLimit;
121}

123//---------------------------------------------------------------------------
124/*
125 flush dirty sectors
126 */
127static int scache_flushSector(cache_set* cache, int index) {
128 if (cache->rec[index].writeDirty) {
129 int ret;
130
131 XPRINTF("scache: flushSector dirty index=%d sector=%u \n", index, cache->rec[index].sector);
132 ret = WRITE_SECTOR(cache->dev, cache->rec[index].sector, cache->sectorBuf + (index * BLOCK_SIZE), BLOCK_SIZE/cache->sectorSize);
133 if (ret < 0) {
134 printf("scache: ERROR writing sector to disk! sector=%u\n", cache->rec[index].sector);
135 return ret;
136 }
137
138 cache->rec[index].writeDirty = 0;
139 }
140 return 1;
141}

143int scache_flushSectors(cache_set* cache) {
144 unsigned int i;
145 int counter = 0;
146
147 XPRINTF("cache: flushSectors devId = %i \n", cache->dev->devId);
148
149 XPRINTF("scache: flushSectors writeFlag=%d\n", cache->writeFlag);
150 //no write operation occured since last flush
151 if (cache->writeFlag==0) {
152 return 0;
153 }
154
155 for (i = 0; i < CACHE_SIZE; i++) {
156 int ret;
157
158 if((ret = scache_flushSector(cache, i)) >= 0)
159 counter ++;
160 else
161 return ret;
162 }
163 cache->writeFlag = 0;
164 return counter;
165}

168int scache_readSector(cache_set* cache, unsigned int sector, void** buf) {
169 int index; //index is given in single sectors not octal sectors
170 int ret;
171 unsigned int alignedSector;
172
173 if (cache != NULL) {
174 XPRINTF("cache: readSector devId = %i %p sector = %u \n", cache->dev->devId, cache, sector);
175 }
176 if (cache == NULL) {
177 printf("cache: devId cache not created \n");
178 return -1;
179 }
180
181#ifdef SCACHE_RECORD_STATS
182 cache->cacheAccess ++;
183#endif
184 index = getIndexRead(cache, sector);
185 XPRINTF("cache: indexRead=%i \n", index);
186 if (index >= 0) { //sector found in cache
187#ifdef SCACHE_RECORD_STATS
188 cache->cacheHits ++;
189#endif
190 *buf = cache->sectorBuf + (index * cache->sectorSize);
191 XPRINTF("cache: hit and done reading sector \n");
192
193 return cache->sectorSize;
194 }
195
196 //compute alignedSector - to prevent storage of duplicit sectors in slots
197 alignedSector = (sector/cache->indexLimit)*cache->indexLimit;
198 index = getIndexWrite(cache, alignedSector);
199 XPRINTF("cache: indexWrite=%i slot=%d alignedSector=%u\n", index, index / cache->indexLimit, alignedSector);
200 ret = READ_SECTOR(cache->dev, alignedSector, cache->sectorBuf + (index * cache->sectorSize), BLOCK_SIZE/cache->sectorSize);
201
202 if (ret < 0) {
203 printf("scache: ERROR reading sector from disk! sector=%u\n", alignedSector);
204 return ret;
205 }
206 *buf = cache->sectorBuf + (index * cache->sectorSize) + ((sector%cache->indexLimit) * cache->sectorSize);
207 XPRINTF("cache: done reading physical sector \n");
208
209 //write precaution
210 //cache->flushCounter++;
211 //if (cache->flushCounter == FLUSH_TRIGGER) {
212 //scache_flushSectors(cache);
213 //}
214
215 return cache->sectorSize;
216}

283void scache_invalidate(cache_set* cache, unsigned int sector, int count) {
284 int i;
285
286 XPRINTF("cache: invalidate devId = %i sector = %u count = %d \n", cache->dev->devId, sector, count);
287
288 for(i = 0; i < count; i++, sector++)
289 {
290 int index; //index is given in single sectors not octal sectors
291
292 index = getSlot(cache, sector);
293 if (index >= 0) { //sector found in cache. Write back and invalidate the block it belongs to.
294 scache_flushSector(cache, index);
295
296 cache->rec[index].sector = 0xFFFFFFF0;
297 cache->rec[index].tax = 0;
298 cache->rec[index].writeDirty = 0;
299 }
300 }

304cache_set* scache_init(mass_dev* dev, int sectSize)
305{
306 cache_set* cache;
307 XPRINTF("cache: init devId = %i sectSize = %u \n", dev->devId, sectSize);
308
309 cache = malloc(sizeof(cache_set));
310 if (cache == NULL) {
311 printf("scache init! Sector cache: can't alloate cache!\n");
312 return NULL;
313 }
314
315 XPRINTF("scache init! \n");
316 cache->dev = dev;
317
318 cache->sectorBuf = (unsigned char*) malloc(BLOCK_SIZE * CACHE_SIZE);
319 if (cache->sectorBuf == NULL) {
320 printf("Sector cache: can't alloate memory of size:%d \n", BLOCK_SIZE * CACHE_SIZE);
321 free(cache);
322 return NULL;
323 }
324 XPRINTF("Sector cache: allocated memory at:%p of size:%d \n", cache->sectorBuf, BLOCK_SIZE * CACHE_SIZE);
325
326 //added by Hermes
327 cache->sectorSize = sectSize;
328 cache->indexLimit = BLOCK_SIZE/cache->sectorSize; //number of sectors per 1 cache slot
329#ifdef SCACHE_RECORD_STATS
330 cache->cacheAccess = 0;
331 cache->cacheHits = 0;
332#endif
333 initRecords(cache);
334 return cache;
335}
