173static void rpc_server_thread(void *arg)
174{
175 (void)arg;
176
177 SifInitRpc(0);
178
179 printf("audsrv: creating rpc server\n");
180
181 SifSetRpcQueue(&qd, GetThreadId());
182 SifRegisterRpc(&sd0, AUDSRV_IRX, (void *)rpc_command, rpc_buffer, NULL, NULL, &qd);
183 SifRpcLoop(&qd);
184}

186int initialize_rpc_thread()
187{
188 int rpc_tid;
189
190 rpc_tid = create_thread(rpc_server_thread, 40, 0);
191 printf("audsrv: rpc server thread 0x%x started\n", rpc_tid);
192 return 0;
193}
