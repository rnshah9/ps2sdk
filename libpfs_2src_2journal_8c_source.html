34int pfsJournalChecksum(void *header)
35{
36 u32 *ptr=(u32 *)header;
37 u32 sum=0;
38 int i;
39 for(i=2; i < 256; i++)
40 sum+=ptr[i];
41 return sum & 0xFFFF;
42}

44void pfsJournalWrite(pfs_mount_t *pfsMount, pfs_cache_t *clink, u32 pfsCacheNumBuffers)
45{
46 u32 i=0;
47 u32 logSector=2;
48
49#ifdef PFS_SUPPORT_BHDD
50 if (strcmp(pfsMount->blockDev->devName, "bhdd") == 0)
51 return;
52#endif
53
54 for(i=0; i <pfsCacheNumBuffers;i++)
55 {
56 if((clink[i].flags & PFS_CACHE_FLAG_DIRTY) && clink[i].pfsMount == pfsMount) {
57 if(clink[i].flags & (PFS_CACHE_FLAG_SEGD|PFS_CACHE_FLAG_SEGI))
58 clink[i].u.inode->checksum=pfsInodeCheckSum(clink[i].u.inode);
59 pfsJournalBuf.log[pfsJournalBuf.num].sector = clink[i].block << pfsBlockSize;
60 pfsJournalBuf.log[pfsJournalBuf.num].sub = clink[i].sub;
61 pfsJournalBuf.log[pfsJournalBuf.num].logSector = logSector;
62 pfsJournalBuf.num+=1;
63 }
64 logSector+=2;
65 }
66
67 if(pfsMount->blockDev->transfer(pfsMount->fd, clink->u.inode, 0,
68 (pfsMount->log.number << pfsMount->sector_scale) + 2, pfsCacheNumBuffers*2,
69 PFS_IO_MODE_WRITE)>=0)
70 pfsJournalFlush(pfsMount);
71}

73int pfsJournalReset(pfs_mount_t *pfsMount)
74{
75 int rv;
76
77#ifdef PFS_SUPPORT_BHDD
78 if (strcmp(pfsMount->blockDev->devName, "bhdd") == 0)
79 return 0;
80#endif
81
82 memset(&pfsJournalBuf, 0, sizeof(pfs_journal_t));
83 pfsJournalBuf.magic=PFS_JOUNRNAL_MAGIC;
84
85 pfsMount->blockDev->flushCache(pfsMount->fd);
86
87 rv = pfsMount->blockDev->transfer(pfsMount->fd, &pfsJournalBuf, 0,
88 (pfsMount->log.number << pfsMount->sector_scale), 2, PFS_IO_MODE_WRITE);
89
90 pfsMount->blockDev->flushCache(pfsMount->fd);
91 return rv;
92}

121int pfsJournalRestore(pfs_mount_t *pfsMount)
122{
123 int rv;
124 int result;
125 pfs_cache_t *clink;
126 u32 i;
127
128#ifdef PFS_SUPPORT_BHDD
129 if (strcmp(pfsMount->blockDev->devName, "bhdd") == 0)
130 return 0;
131#endif
132
133 // Read journal buffer from disk
134 rv = pfsMount->blockDev->transfer(pfsMount->fd, &pfsJournalBuf, 0,
135 (pfsMount->log.number << pfsMount->sector_scale), 2, PFS_IO_MODE_READ);
136
137 if(rv || (pfsJournalBuf.magic != PFS_JOUNRNAL_MAGIC) ||
138 (pfsJournalBuf.checksum != (u16)pfsJournalChecksum(&pfsJournalBuf)))
139 {
140 PFS_PRINTF(PFS_DRV_NAME": Error: cannot read log/invalid log\n");
141 return pfsJournalReset(pfsMount);
142 }
143
144 if(pfsJournalBuf.num == 0)
145 {
146 return pfsJournalReset(pfsMount);
147 }
148
149 clink = pfsCacheAllocClean(&result);
150 if(!clink)
151 return result;
152
153 for(i = 0; i < pfsJournalBuf.num; i++)
154 {
155 PFS_PRINTF(PFS_DRV_NAME": Log overwrite %d:%08lx\n", pfsJournalBuf.log[i].sub, pfsJournalBuf.log[i].sector);
156
157 // Read data in from log section on disk into cache buffer
158 rv = pfsMount->blockDev->transfer(pfsMount->fd, clink->u.data, 0,
159 (pfsMount->log.number << pfsMount->sector_scale) + pfsJournalBuf.log[i].logSector, 2,
160 PFS_IO_MODE_READ);
161
162 // Write from cache buffer into destination location on disk
163 if(!rv)
164 pfsMount->blockDev->transfer(pfsMount->fd, clink->u.data, pfsJournalBuf.log[i].sub,
165 pfsJournalBuf.log[i].sector, 2, PFS_IO_MODE_WRITE);
166 }
167
168 pfsCacheFree(clink);
169 return pfsJournalReset(pfsMount);
170}
