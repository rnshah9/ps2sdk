47static int part_isPartitionRecordInvalid(struct block_device* bd, part_raw_record* raw, part_record* rec)
48{
49 M_DEBUG("%s\n", __func__);
50
51 rec->sid = raw->sid;
52 rec->start = getUI32(raw->startLBA);
53 rec->count = getUI32(raw->size);
54
55 if (rec->sid != 0x00) { /* Windows appears to check if the start LBA is not 0 and whether the start LBA is within the disk.
56 There may be checks against the size, but I didn't manage to identify a pattern.
57 If the disk has no partition table (i.e. disks with "removable" media), then this check is also one safeguard. */
58 if ((rec->start == 0) || (rec->start >= bd->sectorCount))
59 return 1;
60 }
61
62 return 0;
63}

122//---------------------------------------------------------------------------
123int part_connect(struct block_device* bd)
124{
125 part_table partTable;
126 unsigned int i;
127 int parts;
128 int rval = -1;
129
130 M_DEBUG("%s\n", __func__);
131
132 // Filter for supported partition IDs:
133 // - First sector of disk can be MBR partition
134 // - 0x05 = extended MBR partition with CHS addressing
135 // - 0x0f = extended MBR partition with LBA addressing
136 if (bd->sectorOffset != 0 &&
137 bd->parId != 0x05 &&
138 bd->parId != 0x0f)
139 return rval;
140
141 if ((parts = part_getPartitionTable(bd, &partTable)) <= 0)
142 return rval;
143
144 for (i = 0; i < parts; i++) {
145 if (partTable.record[i].sid != 0x00) {
146 M_DEBUG("mount partition %d id %02x\n", i, partTable.record[i].sid);
147 part_create(bd, &partTable.record[i], i);
148 rval = 0;
149 }
150 }
151
152 return rval;
153}
