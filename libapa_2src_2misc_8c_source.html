32void *apaAllocMem(int size)
33{
34#ifdef _IOP
35 int intrStat;
36 void *mem;
37
38 CpuSuspendIntr(&intrStat);
39 mem = AllocSysMemory(ALLOC_FIRST, size, NULL);
40 if(mem == NULL)
41 APA_PRINTF(APA_DRV_NAME": error: out of memory\n");
42 CpuResumeIntr(intrStat);
43
44 return mem;
45#else
46 return malloc(size);
47#endif
48}

63int apaGetTime(apa_ps2time_t *tm)
64{
65#ifdef _IOP
66 int i;
67 sceCdCLOCK cdtime;
68 static apa_ps2time_t timeBuf={
69 0, 7, 6, 5, 4, 3, 2000 // used if can not get time...
70 };
71
72 for(i = 0; i < 20; i++)
73 {
74 int ret;
75
76 ret = sceCdReadClock(&cdtime);
77
78 if(ret!=0 && cdtime.stat==0)
79 {
80 timeBuf.sec=btoi(cdtime.second);
81 timeBuf.min=btoi(cdtime.minute);
82 timeBuf.hour=btoi(cdtime.hour);
83 timeBuf.day=btoi(cdtime.day);
84 timeBuf.month=btoi(cdtime.month & 0x7F); //The old CDVDMAN sceCdReadClock() function does not automatically file off the highest bit.
85 timeBuf.year=btoi(cdtime.year) + 2000;
86 break;
87 } else {
88 if(!(cdtime.stat & 0x80))
89 break;
90 }
91
92 DelayThread(100000);
93 }
94
95 memcpy(tm, &timeBuf, sizeof(apa_ps2time_t));
96#else
97 time_t rawtime;
98 struct tm timeinfo;
99 time(&rawtime);
100 // Convert to JST
101 rawtime += (-9 * 60 * 60);
102 gmtime_r(&rawtime, &timeinfo);
103
104 tm->sec = timeinfo.tm_sec;
105 tm->min = timeinfo.tm_min;
106 tm->hour = timeinfo.tm_hour;
107 tm->day = timeinfo.tm_mday;
108 tm->month = timeinfo.tm_mon + 1;
109 tm->year = timeinfo.tm_year + 1900;
110#endif
111
112 return 0;
113}

115int apaGetIlinkID(u8 *idbuf)
116{
117#ifdef _IOP
118 u32 stat;
119 int i;
120
121 for(i = 0; ; i++)
122 {
123 stat=0;
124 memset(idbuf, 0, 32);
125 if((sceCdRI(idbuf, &stat) != 0) && (stat == 0))
126 {
127 return 0;
128 }
129
130 if(i >= 20)
131 break;
132
133 DelayThread(100000);
134 }
135
136 APA_PRINTF(APA_DRV_NAME": Error: cannot get id\n");
137 return -EIO;
138#else
139 memset(idbuf, 0, 32);
140 return 0;
141#endif
142}
