202typedef struct { // size = 384
203 char magic[28]; // Superblock magic, on PS2 MC : "Sony PS2 Memory Card Format "
204 u8 version[12]; // Version number of the format used, 1.2 indicates full support for bad_block_list
205 s16 pagesize; // size in bytes of a memory card page
206 u16 pages_per_cluster; // number of pages in a cluster
207 u16 blocksize; // number of pages in an erase block
208 u16 unused; // unused
209 u32 clusters_per_card; // total size in clusters of the memory card
210 u32 alloc_offset; // Cluster offset of the first allocatable cluster. Cluster values in the FAT and directory entries are relative to this. This is the cluster immediately after the FAT
211 u32 alloc_end; // The cluster after the highest allocatable cluster. Relative to alloc_offset. Not used
212 u32 rootdir_cluster; // First cluster of the root directory. Relative to alloc_offset. Must be zero
213 u32 backup_block1; // Erase block used as a backup area during programming. Normally the the last block on the card, it may have a different value if that block was found to be bad
214 u32 backup_block2; // This block should be erased to all ones. Normally the the second last block on the card
215 u8 unused2[8];
216 u32 ifc_list[32]; // List of indirect FAT clusters. On a standard 8M card there's only one indirect FAT cluster
217 int bad_block_list[32]; // List of erase blocks that have errors and shouldn't be used
218 u8 cardtype; // Memory card type. Must be 2, indicating that this is a PS2 memory card
219 u8 cardflags; // Physical characteristics of the memory card
220 u16 unused3;
221 u32 cluster_size;
222 u32 FATentries_per_cluster;
223 u32 clusters_per_block;
224 int cardform;
225 u32 rootdir_cluster2;
226 u32 unknown1;
227 u32 unknown2;
228 u32 max_allocatable_clusters;
229 u32 unknown3;
230 u32 unknown4;
231 int unknown5;
232} MCDevInfo;
