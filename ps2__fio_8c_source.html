102static inline int fdh_getfd(int devtype,int fd)
103{
104 int count = 1;
105 while ((count < FDH_MAX))
106 {
107 if (fd_info_list[count].used == 0)

109 fd_info_list[count].used = 1;

111 fd_info_list[count].devtype = devtype;
112 return count;
113 }
114 count++;
115 }
116 return -1;

147static inline int convmode_to_iomanx(int stat)
148{
149 int mode = 0;
150
151 if (FIO_SO_ISLNK(stat)) mode |= FIO_S_IFLNK; // Symbolic link
152 if (FIO_SO_ISREG(stat)) mode |= FIO_S_IFREG; // regular file
153 if (FIO_SO_ISDIR(stat)) mode |= FIO_S_IFDIR; // directory
154
155 if (((stat) & FIO_SO_IROTH) == FIO_SO_IROTH) // read
156 mode |= FIO_S_IRUSR | FIO_S_IRGRP | FIO_S_IROTH;
157 if (((stat) & FIO_SO_IWOTH) == FIO_SO_IWOTH) // write
158 mode |= FIO_S_IWUSR | FIO_S_IWGRP | FIO_S_IWOTH;
159 if (((stat) & FIO_SO_IXOTH) == FIO_SO_IXOTH) // execute
160 mode |= FIO_S_IXUSR | FIO_S_IXGRP | FIO_S_IXOTH;
161
162 return mode;
163}

226int ps2netfs_recv_bytes(int sock, char *buf, int bytes)
227{
228 int left;
229
230 left = bytes;
231
232 while (left > 0) {
233 int len;
234
235 len = recv(sock, &buf[bytes - left], left, 0);
236 if (len < 0) {
237 dbgprintf("ps2netfs: recv_bytes error!! (%d)\n",len);
238 return -1;
239 }
240 if (len == 0)
241 { // recv is blocking, so 0 means error
242 dbgprintf("ps2netfs: recv_bytes - disconnected\n");
243 return -2;
244 }
245 left -= len;
246 }
247 return bytes;
248}
