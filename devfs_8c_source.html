106const char *devfs_subdev_to_str(int subdev)
107
108{
109 static char name[4]; /* Supports up to 999 subdevices */
110 int loop;
111
112 if((subdev > 999) || (subdev < 0))
113 {
114 name[0] = 0;
115 }
116 else
117 {
118 loop = 0;
119 name[loop++] = ((subdev / 100) % 10) + '0';
120 name[loop++] = ((subdev / 10) % 10) + '0';
121 name[loop++] = (subdev % 10) + '0';
122 name[loop++] = 0;
123 }
124
125 loop = 0;
126 if(name[0] == '0') loop = 1;
127 if((name[0] == '0') && (name[1] == '0')) loop = 2;
128
129 return &name[loop];
130}

140int devfs_fill_dirent(iox_dirent_t *dirent, int devno)
141
142{
143 devfs_device_t *dev_scan;
144 int curr_no = 0;
145 u32 subdev_loop;
146
147 dev_scan = root_device;
148
149 while(dev_scan != NULL)
150 {
151 if((curr_no + dev_scan->subdev_count) > devno)
152 {
153 for(subdev_loop = 0; subdev_loop < DEVFS_MAX_SUBDEVS; subdev_loop++)
154 {
155 if(dev_scan->subdevs[subdev_loop].valid)
156 {
157 if(curr_no == devno)
158 {
159 break;
160 }
161 curr_no++;
162 }
163 }
164
165 if(subdev_loop < DEVFS_MAX_SUBDEVS)
166 {
167 memset(dirent, 0, sizeof(iox_dirent_t));
168 dirent->stat.size = dev_scan->subdevs[subdev_loop].extent.loc32[0];
169 dirent->stat.hisize = dev_scan->subdevs[subdev_loop].extent.loc32[1];
170 dirent->stat.mode = FIO_S_IFREG;
171 if(dev_scan->subdevs[subdev_loop].mode & DEVFS_MODE_R)
172 {
173 dirent->stat.mode |= FIO_S_IRUSR;
174 }
175 if(dev_scan->subdevs[subdev_loop].mode & DEVFS_MODE_W)
176 {
177 dirent->stat.mode |= FIO_S_IWUSR;
178 }
179 dirent->name[0] = 0;
180 strcpy(dirent->name, dev_scan->node.name);
181 strcat(dirent->name, devfs_subdev_to_str(subdev_loop));
182
183 return strlen(dirent->name);
184 }
185 }
186
187 curr_no += dev_scan->subdev_count;
188
189 dev_scan = dev_scan->forw;
190 }
191
192 return 0;
193}

201devfs_device_t *devfs_create_device(const devfs_node_t *node)
202
203{
204 devfs_device_t *dev;
205
206 if((node == NULL) || (node->name == NULL))
207 {
208 return NULL;
209 }
210
211 dev = AllocSysMemory(ALLOC_FIRST, sizeof(devfs_device_t), NULL);
212 if(dev == NULL)
213 {
214 printf("create_device: Alloc Failed\n");
215 return NULL;
216 }
217
218 memset(dev, 0, sizeof(devfs_device_t));
219 memcpy(&dev->node, node, sizeof(devfs_node_t));
220 if(dev->node.name != NULL)
221 {
222 dev->node.name = AllocSysMemory(ALLOC_FIRST, strlen(node->name) + 1, NULL);
223 if(dev->node.name == NULL)
224 {
225 FreeSysMemory(dev);
226 return NULL;
227 }
228 memcpy(dev->node.name, node->name, strlen(node->name) + 1);
229 }
230
231 if(dev->node.desc != NULL)
232 {
233 dev->node.desc = AllocSysMemory(ALLOC_FIRST, strlen(node->desc) + 1, NULL);
234 if(dev->node.name == NULL)
235 {
236 FreeSysMemory(dev->node.name);
237 FreeSysMemory(dev);
238 return NULL;
239 }
240 memcpy(dev->node.desc, node->desc, strlen(node->desc) + 1);
241 }
242
243 return dev;
244}

367#ifdef USE_IOMAN
368int devfs_open(iop_file_t *file, const char *name, int mode)
369#else
370int devfs_open(iop_file_t *file, const char *name, int mode, int unused)
371#endif
372{
373 devfs_device_t *dev;
374 int fn_offset = 0;
375
376 (void)unused;
377
378 //printf("devfs_open file=%p name=%s mode=%d\n", file, name, mode);
379 if(name == NULL)
380 {
381 M_PRINTF("open: Name is NULL\n");
382 return -1;
383 }
384
385 if((name[0] == '\\') || (name[0] == '/'))
386 {
387 fn_offset = 1;
388 }
389
390 dev = devfs_find_devicename(name + fn_offset);
391
392 if(dev != NULL)
393 {
394 int loop;
395 char *endp;
396 int subdev;
397 int name_len;
398
399 name_len = strlen(dev->node.name);
400
401 if(name_len == strlen(name)) /* If this is has not got a subnumber */
402 {
403 M_PRINTF("open: No subdevice number in filename %s\n", name);
404 return -1;
405 }
406
407 subdev = strtoul(&name[name_len + fn_offset], &endp, 10);
408 if(((subdev == 0) && (name[name_len + fn_offset] != '0'))
409 || (subdev >= DEVFS_MAX_SUBDEVS))
410 /* Invalid number */
411 {
412 M_PRINTF("open: Invalid subdev number %d\n", subdev);
413 return -1;
414 }
415
416 if(*endp)
417 /* Extra charactes after filename */
418 {
419 M_PRINTF("open: Invalid filename\n");
420 return -1;
421 }
422
423 if(!dev->subdevs[subdev].valid)
424 /* No subdev */
425 {
426 M_PRINTF("open: No subdev registered\n");
427 return -1;
428 }
429
430 if((dev->subdevs[subdev].mode & DEVFS_MODE_EX)
431 && (dev->subdevs[subdev].open_refcount > 0))
432 /* Already opened in exclusive mode */
433 {
434 M_PRINTF("open: Exclusive subdevice already opened\n");
435 return -1;
436 }
437
438 if(dev->subdevs[subdev].open_refcount == MAX_OPENFILES)
439 {
440 M_PRINTF("open: Reached open file limit for sub device\n");
441 }
442
443 /* Tried to open read but not allowed */
444 if(((mode & O_RDONLY) || ((mode & O_RDWR) == O_RDWR))
445 && !(dev->subdevs[subdev].mode & DEVFS_MODE_R))
446 {
447 M_PRINTF("open: Read mode requested but not permitted\n");
448 return -1;
449 }
450
451 if(((mode & O_WRONLY) || ((mode & O_RDWR) == O_RDWR))
452 && !(dev->subdevs[subdev].mode & DEVFS_MODE_W))
453 {
454 M_PRINTF("open: Write mode requested but not permitted\n");
455 return -1;
456 }
457
458 file->privdata = AllocSysMemory(ALLOC_FIRST, sizeof(ioman_data_t), NULL);
459 if(file->privdata == NULL)
460 {
461 M_PRINTF("open: Allocation failure\n");
462 return -1;
463 }
464
465 ((ioman_data_t *) file->privdata)->hDev = dev->hDev;
466 ((ioman_data_t *) file->privdata)->subdev = subdev;
467 ((ioman_data_t *) file->privdata)->loc.loc64 = 0;
468 ((ioman_data_t *) file->privdata)->dev = dev;
469 ((ioman_data_t *) file->privdata)->mode = mode;
470
471 dev->subdevs[subdev].open_refcount++;
472 loop = 0;
473 while((loop < MAX_OPENFILES) && (dev->subdevs[subdev].open_files[loop]))
474 {
475 loop++;
476 }
477 if(loop == MAX_OPENFILES)
478 {
479 FreeSysMemory(file->privdata);
480 M_PRINTF("open: Inconsistency between number of open files and available slot\n");
481 return -1;
482 }
483
484 dev->subdevs[subdev].open_files[loop] = file->privdata;
485 dev->open_refcount++;
486 M_PRINTF("open: Opened device %s subdev %d\n", dev->node.name, subdev);
487 }
488 else
489 {
490 M_PRINTF("open: Couldn't find the device\n");
491 return -1;
492 }
493
494 return 0;
495}

502int devfs_close(iop_file_t *file)
503
504{
505 ioman_data_t *data = (ioman_data_t *) file->privdata;
506
507 if((data) && (data->hDev != INVALID_HDEV)) /* Invalid HDEV is a dir listing */
508 {
509 devfs_device_t *dev;
510 dev = data->dev;
511
512 if(dev != NULL)
513 {
514 int loop;
515
516 if(dev->subdevs[data->subdev].open_refcount > 0)
517 {
518 dev->subdevs[data->subdev].open_refcount--;
519 }
520 if(dev->open_refcount > 0)
521 {
522 dev->open_refcount--;
523 }
524
525 loop = 0;
526 while((loop < MAX_OPENFILES)
527 && (dev->subdevs[data->subdev].open_files[loop] != data))
528 {
529 loop++;
530 }
531
532 if(loop != MAX_OPENFILES)
533 {
534 dev->subdevs[data->subdev].open_files[loop] = NULL;
535 }
536 else
537 {
538 M_PRINTF("close: Could not find opened file\n");
539 }
540
541 M_PRINTF("close: Closing device %s subdev %d\n", dev->node.name, data->subdev);
542 FreeSysMemory(data);
543 }
544 else
545 {
546 FreeSysMemory(data);
547 file->privdata = NULL;
548 return -1;
549 }
550 }
551 else
552 {
553 return -1;
554 }
555
556 return 0;
557}

565int devfs_ioctl(iop_file_t *file, int cmd, void *args)
566
567{
568 devfs_device_t *dev;
569 ioman_data_t *data = (ioman_data_t *) file->privdata;
570
571 if(cmd == DEVFS_IOCTL_GETDESC)
572 {
573 return -1; /* ioctl cannot support this call */
574 }
575
576 data = file->privdata;
577 if((data) && (data->hDev != INVALID_HDEV))
578 {
579 dev = data->dev;
580 if(dev == NULL) /* Device has been closed from under us */
581 {
582 /* Delete data to try and prevent a memory leak */
583 FreeSysMemory(data);
584 file->privdata = NULL;
585 return -1;
586 }
587
588 if(dev->node.ioctl != NULL)
589 {
590 devfs_info_t dev_info;
591
592 dev_info.data = dev->subdevs[data->subdev].data;
593 dev_info.subdev = data->subdev;
594 dev_info.mode = data->mode;
595 dev_info.loc = data->loc;
596 return dev->node.ioctl(&dev_info, DEVFS_IOCTL_TYPE_1, cmd, args, 0, NULL, 0);
597 }
598
599 }
600
601 return -1;
602}

760int devfs_lseek(iop_file_t *file, long loc, int whence)
761{
762 devfs_device_t *dev;
763 ioman_data_t *data = (ioman_data_t *) file->privdata;
764
765 //printf("devfs_lseek file=%p loc=%ld whence=%d\n", file, loc, whence);
766 data = file->privdata;
767 if((data) && (data->hDev != INVALID_HDEV))
768 {
769 dev = data->dev;
770 if(dev == NULL) /* Device has been closed from under us */
771 {
772 /* Delete data to try and prevent a memory leak */
773 FreeSysMemory(data);
774 file->privdata = NULL;
775 return -1;
776 }
777
778 switch(whence)
779 {
780 case SEEK_SET: if(loc > 0)
781 data->loc.loc64 = (u64) loc;
782 break;
783 case SEEK_CUR: if(loc > 0)
784 data->loc.loc64 += (u64) loc;
785 else if(((s64) -loc) < data->loc.loc64)
786 data->loc.loc64 += (s64) loc;
787 break;
788 case SEEK_END: if((loc > 0)
789 && (dev->subdevs[data->subdev].extent.loc64 >= (u64) loc))
790 data->loc.loc64 = dev->subdevs[data->subdev].extent.loc64 - (u64) loc;
791 break;
792 default: return -1;
793 }
794 }
795 else
796 {
797 return -1;
798 }
799
800 return data->loc.loc32[0];
801}
