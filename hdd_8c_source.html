101apa_cache_t *hddAddPartitionHere(s32 device, const apa_params_t *params, u32 *emptyBlocks,
102 u32 sector, int *err)
103{
104 apa_cache_t *clink_this;
105 apa_cache_t *clink_next;
106 apa_cache_t *clink_new;
107 apa_header_t *header;
108 u32 i;
109 u32 tmp, some_size, part_end;
110 u32 tempSize;
111
112 // walk empty blocks in case can use one :)
113 for(i=0;i< 32;i++)
114 {
115 if((u32)(1 << i) >= params->size && emptyBlocks[i]!=0)
116 return apaInsertPartition(device, params, emptyBlocks[i], err);
117 }
118 clink_this=apaCacheGetHeader(device, sector, APA_IO_MODE_READ, err);
119 header=clink_this->header;
120 part_end=header->start+header->length;
121 some_size=(part_end%params->size);
122 tmp = some_size ? params->size - some_size : 0;
123
124 if(hddDevices[device].totalLBA < (part_end + params->size + tmp)
125 //Non-SONY: when dealing with large disks, this check may overflow (therefore, check for overflows!).
126 || (part_end < sector))
127 {
128 *err=-ENOSPC;
129 apaCacheFree(clink_this);
130 return NULL;
131 }
132
133 if((clink_next=apaCacheGetHeader(device, 0, APA_IO_MODE_READ, err))==NULL)
134 {
135 apaCacheFree(clink_this);
136 return NULL;
137 }
138
139 tempSize=params->size;
140 while(part_end%params->size)
141 {
142 tempSize=params->size>>1;
143 while(0x3FFFF<tempSize)
144 {
145 if(!(part_end%tempSize))
146 {
147 clink_new=apaRemovePartition(device, part_end, 0,
148 clink_this->header->start, tempSize);
149 clink_this->header->next=part_end;
150 clink_this->flags|=APA_CACHE_FLAG_DIRTY;
151 clink_next->header->prev=clink_new->header->start;
152 part_end+=tempSize;
153 clink_next->flags|=APA_CACHE_FLAG_DIRTY;
154 apaCacheFlushAllDirty(device);
155 apaCacheFree(clink_this);
156 clink_this=clink_new;
157 break;
158 }
159 tempSize>>=1;
160 }
161 }
162 if((clink_new=apaFillHeader(device, params, part_end, 0, clink_this->header->start,
163 params->size, err))!=NULL)
164 {
165 clink_this->header->next=part_end;
166 clink_this->flags|=APA_CACHE_FLAG_DIRTY;
167 clink_next->header->prev=clink_new->header->start;
168 clink_next->flags|=APA_CACHE_FLAG_DIRTY;
169 apaCacheFlushAllDirty(device);
170 }
171 apaCacheFree(clink_this);
172 apaCacheFree(clink_next);
173 return clink_new;
174}

197int _start(int argc, char **argv)
198{
199 int i, ret;
200 char *input;
201 int cacheSize=3;
202 apa_ps2time_t tm;
203 ata_devinfo_t *hddInfo;
204
205 printStartup();
206
207 if((input=strrchr(argv[0], '/')))
208 input++;
209 else
210 input=argv[0];
211
212 argc--; argv++;
213 while(argc)
214 {
215 if(argv[0][0] != '-')
216 break;
217 if(strcmp("-o", argv[0])==0)
218 {
219 argc--; argv++;
220 if(!argc)
221 return inputError(input);
222 i=strtol(argv[0], 0, 10);
223 if(i-1<32)
224 apaMaxOpen=i;
225 }
226 else if(strcmp("-n", argv[0])==0)
227 {
228 argc--; argv++;
229 if(!argc)
230 return inputError(input);
231 i=strtol(*argv, 0, 10);
232 if(cacheSize<i)
233 cacheSize=i;
234 }
235 argc--; argv++;
236 }
237
238 APA_PRINTF(APA_DRV_NAME": max open = %d, %d buffers\n", apaMaxOpen, cacheSize);
239 if(dev9RegisterShutdownCb(0, &hddShutdownCb) != 0)
240 {
241 APA_PRINTF(APA_DRV_NAME": error: dev9 may not be resident.\n");
242 return hddInitError();
243 }
244
245 if(apaGetTime(&tm) != 0)
246 {
247 APA_PRINTF(APA_DRV_NAME": error: could not get date.\n");
248 return hddInitError();
249 }
250
251 APA_PRINTF(APA_DRV_NAME": %02d:%02d:%02d %02d/%02d/%d\n",
252 tm.hour, tm.min, tm.sec, tm.month, tm.day, tm.year);
253 for(i=0;i < 2;i++)
254 {
255 if(!(hddInfo=ata_get_devinfo(i)))
256 {
257 APA_PRINTF(APA_DRV_NAME": Error: ata initialization failed.\n");
258 return hddInitError();
259 }
260 if(hddInfo->exists!=0 && hddInfo->has_packet==0)
261 {
262 hddDevices[i].status--;
263 hddDevices[i].totalLBA=hddInfo->total_sectors;
264 hddDevices[i].partitionMaxSize=apaGetPartitionMax(hddInfo->total_sectors);
265 if(unlockDrive(i)==0)
266 hddDevices[i].status--;
267 APA_PRINTF(APA_DRV_NAME": disk%d: 0x%08lx sectors, max 0x%08lx\n", i,
268 hddDevices[i].totalLBA, hddDevices[i].partitionMaxSize);
269 }
270 }
271 hddFileSlots=apaAllocMem(apaMaxOpen*sizeof(hdd_file_slot_t));
272 ret = (hddFileSlots == NULL) ? -ENOMEM : 0;
273 if(ret != 0)
274 {
275 APA_PRINTF(APA_DRV_NAME": error: file descriptor initialization failed.\n");
276 return hddInitError();
277 }
278
279 memset(hddFileSlots, 0, apaMaxOpen*sizeof(hdd_file_slot_t));
280
281 if(apaCacheInit(cacheSize) != 0)
282 {
283 APA_PRINTF(APA_DRV_NAME": error: cache buffer initialization failed.\n");
284 return hddInitError();
285 }
286
287 for(i=0;i < 2;i++)
288 {
289 if(hddDevices[i].status<2)
290 {
291 if(apaJournalRestore(i) != 0)
292 {
293 APA_PRINTF(APA_DRV_NAME": error: log check failed.\n");
294 return hddInitError();
295 }
296 if(apaGetFormat(i, &hddDevices[i].format))
297 hddDevices[i].status--;
298 APA_PRINTF(APA_DRV_NAME": drive status %d, format version %08x\n",
299 hddDevices[i].status, hddDevices[i].format);
300 }
301 }
302 DelDrv("hdd");
303 if(AddDrv(&hddFioDev) == 0)
304 {
305#ifdef APA_SUPPORT_BHDD
306 DelDrv("bhdd");
307 if(AddDrv(&bhddFioDev) == 0)
308#endif
309 {
310#if defined(APA_XOSD_VER)
311 APA_PRINTF(APA_DRV_NAME": version %04x driver start. This is XOSD LBA48 VERSION !!!!!!!!!!!\n", IRX_VER(APA_MODVER_MAJOR, APA_MODVER_MINOR));
312#elif defined(APA_OSD_VER)
313 APA_PRINTF(APA_DRV_NAME": version %04x driver start. This is OSD version!\n", IRX_VER(APA_MODVER_MAJOR, APA_MODVER_MINOR));
314#else
315 APA_PRINTF(APA_DRV_NAME": version %04x driver start.\n", IRX_VER(APA_MODVER_MAJOR, APA_MODVER_MINOR));
316#endif
317 return MODULE_RESIDENT_END;
318 }
319 }
320 {
321 APA_PRINTF(APA_DRV_NAME": error: add device failed.\n");
322 return hddInitError();
323 }
324}
