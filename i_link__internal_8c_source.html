34void UBUFThread(void *arg){
35 void (*RequestResponseHandler)(unsigned int header, volatile unsigned int *buffer, unsigned int nQuads);
36 unsigned int data;
37 u32 ef_bits;
38
39 (void)arg;
40
41 while(1){
42 unsigned int DataBufferLevel;
43
44 WaitEventFlag(IntrEventFlag, iLinkEventBusReset|iLinkEventURx, WEF_OR, &ef_bits);
45 ClearEventFlag(IntrEventFlag, ~(iLinkEventBusReset|iLinkEventURx));
46
47 while((DataBufferLevel=ILINKRegisterBase->ubufReceiveLevel)>0){
48 data=ILINKRegisterBase->ubufReceive;
49 iDEBUG_PRINTF("Handling UBUF data... handler: 0x%02lx.\n", (data>>4)&0xF);
50 RequestResponseHandler=RequestHandlers[(data>>4)&0xF];
51 RequestResponseHandler(data, (volatile unsigned int *)&ILINKRegisterBase->ubufReceive, DataBufferLevel-1);
52 }
53 }
54}

117int iLinkResetHW(void){
118 ShutDownDMAChannels();
119
120 /* Turn off, and then turn back on the LINK and PHY. If this is not done, the iLink hardware might not function correctly on some consoles. :( */
121 ILINKRegisterBase->UnknownRegister7C=0x40; /* Shut down. */
122 DelayThread(400000);
123 ILINKRegisterBase->UnknownRegister7C=0;
124
125 /* If the LINK is off, switch it on. */
126 if(!(ILINKRegisterBase->ctrl2&iLink_CTRL2_LPSEn)) ILINKRegisterBase->ctrl2=iLink_CTRL2_LPSEn;
127
128 /* Wait for the clock to stabilize. */
129 while(!(ILINKRegisterBase->ctrl2&iLink_CTRL2_SOK)) DelayThread(50);
130
131 return 0;
132}
