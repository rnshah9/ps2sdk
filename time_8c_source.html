40static int intrOverflow(int ca)
41{
42 (void)ca;
43
44 __time_intr_overflow_count++;
45
46 // A write to the overflow flag will clear the overflow flag
47 // ---------------------------------------------------------
48 *T_MODE |= (1 << 11);
49
50 ExitHandler();
51 return -1;
52}

54void _ps2sdk_time_init(void)
55{
56 *T_MODE = 0x0000; // Disable T_MODE
57
58 if (__time_intr_overflow_id == -1)
59 {
60 __time_intr_overflow_id = AddIntcHandler(INTC_TIM, intrOverflow, 0);
61 EnableIntc(INTC_TIM);
62 }
63
64 // Initialize the timer registers
65 // CLKS: 0x02 - 1/256 of the BUSCLK (0x01 is 1/16th)
66 // CUE: 0x01 - Start/Restart the counting
67 // OVFE: 0x01 - An interrupt is generated when an overflow occurs
68 *T_COUNT = 0;
69 *T_MODE = Tn_MODE(0x02, 0, 0, 0, 0, 0x01, 0, 0x01, 0, 0);
70
71 __time_intr_overflow_count = 0;
72}

74void _ps2sdk_time_deinit(void)
75{
76 *T_MODE = 0x0000; // Stop the timer
77
78 if (__time_intr_overflow_id >= 0)
79 {
80 DisableIntc(INTC_TIM);
81 RemoveIntcHandler(INTC_TIM, __time_intr_overflow_id);
82 __time_intr_overflow_id = -1;
83 }
84
85 __time_intr_overflow_count = 0;
86}
