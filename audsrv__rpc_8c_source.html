228int audsrv_play_audio(const char *chunk, int bytes)
229{
230 int maxcopy;
231 int sent = 0;
232
233 set_error(AUDSRV_ERR_NOERROR);
234 maxcopy = sizeof(sbuff) - sizeof(int);
235 while (bytes > 0)
236 {
237 int copy, copied;
238 int packet_size;
239
240 WaitSema(completion_sema);
241
242 copy = MIN(bytes, maxcopy);
243 sbuff[0] = copy;
244 memcpy(&sbuff[1], chunk, copy);
245 packet_size = copy + sizeof(int);
246 SifCallRpc(&cd0, AUDSRV_PLAY_AUDIO, 0, sbuff, packet_size, sbuff, 1*4, NULL, NULL);
247
248 copied = sbuff[0];
249 SignalSema(completion_sema);
250
251 if (copied < 0)
252 {
253 /* there was an error */
254 set_error(-copied);
255 break;
256 }
257
258 chunk = chunk + copy;
259 bytes = bytes - copy;
260 sent = sent + copied;
261 }
262
263 return sent;
264}

302int audsrv_init()
303{
304 ee_sema_t compSema;
305 ee_thread_t rpcThread;
306 int ret;
307
308 if (initialized)
309 {
310 /* already done */
311 return 0;
312 }
313
314 memset(&cd0, '\0', sizeof(cd0));
315
316 while (1)
317 {
318 if (SifBindRpc(&cd0, AUDSRV_IRX, 0) < 0)
319 {
320 set_error(AUDSRV_ERR_RPC_FAILED);
321 return -1;
322 }
323
324 if (cd0.server != 0)
325 {
326 break;
327 }
328
329 nopdelay();
330 }
331
332 compSema.init_count = 1;
333 compSema.max_count = 1;
334 compSema.option = 0;
335 completion_sema = CreateSema(&compSema);
336 if (completion_sema < 0)
337 {
338 set_error(AUDSRV_ERR_FAILED_TO_CREATE_SEMA);
339 return -1;
340 }
341
342 /* Create RPC server */
343 rpcThread.attr = 0;
344 rpcThread.option = 0;
345 rpcThread.func = &rpc_server_thread;
346 rpcThread.stack = rpc_server_stack;
347 rpcThread.stack_size = sizeof(rpc_server_stack);
348 rpcThread.gp_reg = &_gp;
349 rpcThread.initial_priority = 0x60;
350 rpc_server_thread_id = CreateThread(&rpcThread);
351 StartThread(rpc_server_thread_id, NULL);
352
353 SifCallRpc(&cd0, AUDSRV_INIT, 0, sbuff, 64, sbuff, 64, NULL, NULL);
354 ret = sbuff[0];
355 if (ret != 0)
356 {
357 set_error(ret);
358 return ret;
359 }
360
361 /* initialize IOP heap (for adpcm samples) */
362 SifInitIopHeap();
363
364 set_error(AUDSRV_ERR_NOERROR);
365 return AUDSRV_ERR_NOERROR;
366}

387int audsrv_load_adpcm(audsrv_adpcm_t *adpcm, void *buffer, int size)
388{
389 void* iop_addr;
390 SifDmaTransfer_t sifdma;
391 int id, ret;
392
393 iop_addr = SifAllocIopHeap(size);
394 if (iop_addr == 0)
395 {
396 return -AUDSRV_ERR_OUT_OF_MEMORY;
397 }
398
399 sifdma.src = buffer;
400 sifdma.dest = iop_addr;
401 sifdma.size = size;
402 sifdma.attr = 0;
403
404 /* send by dma */
405 while((id = SifSetDma(&sifdma, 1)) == 0);
406 while(SifDmaStat(id) >= 0);
407
408 WaitSema(completion_sema);
409
410 sbuff[0] = (int)iop_addr;
411 sbuff[1] = size;
412 sbuff[2] = (int)adpcm; /* use as id */
413
414 SifCallRpc(&cd0, AUDSRV_LOAD_ADPCM, 0, sbuff, 12, sbuff, 16, NULL, NULL);
415
416 if(sbuff[0] != 0)
417 {
418 adpcm->buffer = 0;
419 ret = sbuff[0];
420 }
421 else
422 {
423 adpcm->buffer = buffer;
424 adpcm->size = size;
425 adpcm->pitch = sbuff[1];
426 adpcm->loop = sbuff[2];
427 adpcm->channels = sbuff[3];
428 ret = AUDSRV_ERR_NOERROR;
429 }
430
431 SignalSema(completion_sema);
432
433 SifFreeIopHeap(iop_addr);
434
435 return ret;
436}
