55int nanosleep(const struct timespec *req, struct timespec *rem)
56{
57 ee_sema_t sema;
58 s32 sema_id;
59 ps2_clock_t clock_delay, clock_end, clock_real_end;
60 u16 hsync_delay;
61
62 clock_delay = timespec_to_ps2_clock(req);
63 clock_end = ps2_clock() + clock_delay;
64 hsync_delay = clock_delay / iPS2ClockPerHSync;
65
66 if (hsync_delay >= MIN_HSYNC_DELAY) {
67 sema.init_count = 0;
68 sema.max_count = 1;
69 sema.option = 0;
70 sema_id = CreateSema(&sema);
71
72 SetAlarm(hsync_delay, _sleep_waker, &sema_id);
73 WaitSema(sema_id);
74 DeleteSema(sema_id);
75
76 // Correct the HSync delay
77 // NOTE: We're always trying to end btween 0 and 1 HSyncs too short
78 clock_real_end = ps2_clock();
79 if (clock_real_end < (clock_end - iPS2ClockPerHSync)) {
80 // We sleeped too short, so HSyncs must be faster
81 // Correct the HSyncs for next time
82 iPS2ClockPerHSync -= (clock_end - iPS2ClockPerHSync - clock_real_end) / hsync_delay;
83 }
84 else if (clock_real_end > clock_end) {
85 // We sleeped too long, so HSyncs must be slower
86 // Correct the HSyncs for next time
87 iPS2ClockPerHSync += (clock_real_end - clock_end) / (hsync_delay-1);
88 }
89 }
90
91 // Delay whatever time is left
92 while(clock_end > ps2_clock())
93 ;
94
95 if (rem != NULL) {
96 rem->tv_sec = 0;
97 rem->tv_nsec = 0;
98 }
99
100 return 0;
101}
