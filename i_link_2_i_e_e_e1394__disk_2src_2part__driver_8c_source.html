50int part_getPartitionTable(struct SBP2Device* dev, part_table* part)
51{
52 int i;
53 int ret;
54 unsigned char* sbuf;
55
56 ret = READ_SECTOR(dev, 0, sbuf); // read sector 0 - Disk MBR or boot sector
57 if ( ret < 0 )
58 {
59 printf("USBHDFSD: part_getPartitionTable read failed %d!\n", ret);
60 return -1;
61 }
62
63 printf("USBHDFSD: boot signature %X %X\n", sbuf[0x1FE], sbuf[0x1FF]);
64 if (sbuf[0x1FE] == 0x55 && sbuf[0x1FF] == 0xAA)
65 {
66 for ( i = 0; i < 4; i++)
67 {
68 part_raw_record* part_raw;
69
70 part_raw = ( part_raw_record* )( sbuf + 0x01BE + ( i * 16 ) );
71 part_getPartitionRecord(part_raw, &part->record[i]);
72 }
73 return 4;
74 }
75 else
76 {
77 for ( i = 0; i < 4; i++)
78 {
79 part->record[i].sid = 0;;
80 }
81 return 0;
82 }
83}

85//---------------------------------------------------------------------------
86int part_connect(struct SBP2Device* dev)
87{
88 part_table partTable;
89 int count = 0;
90 int i;
91 XPRINTF("USBHDFSD: part_connect devId %i \n", dev->nodeID);
92
93 if (part_getPartitionTable(dev, &partTable) < 0)
94 return -1;
95
96 for ( i = 0; i < 4; i++)
97 {
98 if(
99 partTable.record[ i ].sid == 6 ||
100 partTable.record[ i ].sid == 4 ||
101 partTable.record[ i ].sid == 1 || // fat 16, fat 12
102 partTable.record[ i ].sid == 0x0B ||
103 partTable.record[ i ].sid == 0x0C || // fat 32
104 partTable.record[ i ].sid == 0x0E) // fat 16 LBA
105 {
106 XPRINTF("USBHDFSD: mount partition %d\n", i);
107 if (fat_mount(dev, partTable.record[i].start, partTable.record[i].count) >= 0)
108 count++;
109 }
110 }
111
112 if ( count == 0 )
113 { // no partition table detected
114 // try to use "floppy" option
115 printf("USBHDFSD: mount drive\n");
116 if (fat_mount(dev, 0, dev->maxLBA) < 0)
117 return -1;
118 }
119
120 return 0;
121}
