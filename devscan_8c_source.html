79int devscan_setup(int devtype)
80{
81 ModuleInfo_t *info;
82 iop_device_t **devinfo_table;
83 int i;
84 int count = 0;
85
86 dbgprintf("devscan: setup\n");
87 // clear device list
88 memset(&dev_info_list,0,sizeof(dev_info_list));
89
90 /* do the ioman devices */
91 if ((info = devscan_getmodule(IOPMGR_IOMAN_IDENT)))
92 {
93 /* Find the start of the device info array, in .bss. */
94 devinfo_table = (iop_device_t **)(info->text_start + info->text_size + info->data_size + 0x0c);
95
96 /* The device info table had 16 entries, but some may be empty. Just look at all of them. */
97 for (i = 0; i < DEVSCAN_IOMAX; i++)
98 {
99 if (devinfo_table[i])
100 if ((devinfo_table[i]->type & devtype))
101 {
102 dev_info_list[count].devtype = IOPMGR_DEVTYPE_IOMAN;
103 strncpy(dev_info_list[count].name,devinfo_table[i]->name,255);
104 dev_info_list[count].name[255] = '\0';
105 dev_info_list[count].len = strlen(dev_info_list[count].name);
106 dbgprintf("devscan: ioman '%s'\n",dev_info_list[count].name);
107 count++;
108 }
109 }
110 }
111
112 /* do the iomanx devices */
113 if ((info = devscan_getmodule(IOPMGR_IOMANX_IDENT)))
114 {
115 /* Find the start of the device info array, in .bss. */
116 devinfo_table = (iop_device_t **)(info->text_start + info->text_size + info->data_size);
117
118 /* The device info table had 32 entries, but some may be empty. Just look at all of them. */
119 for (i = 0; i < DEVSCAN_IOXMAX; i++)
120 {
121 if (devinfo_table[i])
122 /* only add iomanx ones here, so must have extended flag set else we get duplication with new iomanx */
123 if ((devinfo_table[i]->type & IOP_DT_FSEXT) && (devinfo_table[i]->type & devtype))
124 {
125 dev_info_list[count].devtype = IOPMGR_DEVTYPE_IOMANX;
126 strncpy(dev_info_list[count].name,devinfo_table[i]->name,255);
127 dev_info_list[count].name[255] = '\0';
128 dev_info_list[count].len = strlen(dev_info_list[count].name);
129 dbgprintf("devscan: iomanx '%s'\n",dev_info_list[count].name);
130 count++;
131 }
132 }
133 }
134 return count;
135}

147int devscan_gettype(char *name)
148{
149 int count = 0;
150 dbgprintf("devscan: gettype '%s'\n",name);
151 while (dev_info_list[count].name[0] != 0)
152 {
153 int ret = strncmp(dev_info_list[count].name,name,dev_info_list[count].len );
154 dbgprintf("'%s'",dev_info_list[count].name);
155 if (!ret)
156 return dev_info_list[count].devtype;
157 count++;
158 }
159 return IOPMGR_DEVTYPE_INVALID;
160}

168int devscan_getdevlist(char *buffer)
169{

173
174 dbgprintf("devscan: getdevlist\n");
175 /* rescan for devices, before returning list */
176 count = devscan_setup(DEVSCAN_MASK);
177
178 /* now convert this list into the right format */
179 for (i=0;i<count;i++)
180 {
181 strcpy(bufptr,dev_info_list[i].name);
182 dbgprintf("devscan: '%s'\n",bufptr);
183 bufptr += strlen(bufptr)+1;
184 }
185 return count;
