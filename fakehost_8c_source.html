219int _start( int argc, char **argv )
220{
221 M_PRINTF( "Copyright (c) 2004 adresd\n" );
222
223 if ( argc != 2 )
224 {
225 M_PRINTF( "HOST requires based argument\n" );
226 return -1;
227 }
228 else
229 {
230 // Copy the base location.
231 strncpy( base, argv[1] ,sizeof(base) - 1);
232 base[sizeof(base) - 1] = '\0';
233
234 M_PRINTF( "redirecting '%s:' to '%s'\n",FS_REPNAME,base);
235 fd_global = 1;
236
237 driver.name = FS_REPNAME;
238 driver.type = 16;
239 driver.version = 1;
240 driver.desc = "host redirection driver";
241 driver.ops = &functions;
